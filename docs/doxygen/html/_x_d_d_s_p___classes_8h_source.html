104template <int BufferCount>
105class OutputBuffer final : public Parameters::ParameterListener
106{
107 Parameters& dspParam;
108
109 std::vector<SampleType> buffer;
110
111 SampleType& get(int channel, int index)
112 {
113 dsp_assert(channel >= 0 && channel < BufferCount);
114 return buffer[channel*dspParam.bufferSize() + index];
115 }
116
117public:
118 static constexpr int Count = BufferCount;
119
120 OutputBuffer(Parameters &p) :
121 Parameters::ParameterListener(p),
122 dspParam(p)
123 { updateBufferSize(p.bufferSize()); }
124
125 OutputBuffer(const OutputBuffer<Count> &rhs) :
126 Parameters::ParameterListener(rhs.dspParam),
127 dspParam(rhs.dspParam),
128 buffer(rhs.buffer)
129 {}
130
131 virtual void updateBufferSize(int bs) override
132 {
133 buffer.resize(bs*Count, 0.);
134 buffer.shrink_to_fit();
135 }
136
137 SampleType* operator[](int channel)
138 { return &get(channel, 0); }
139
140 SampleType& operator()(int channel, int index)
141 { return get(channel, index); }
142
143 SampleType& operator()(int index)
144 { return get(0, index); }
145
146 void reset()
147 { buffer.assign(buffer.size(), 0.); }
148};

276template <class Derived, int StepSize = INT_MAX>
277class Component : public ComponentBaseClass
278{
279 Derived &THIS;
280
281public:
282 Component() :
283 THIS(static_cast<Derived&>(*this))
284 {}
285
290 void reset()
291 {}
292
301 int startProcess(int startPoint, int sampleCount)
302 { return std::min(sampleCount, StepSize); }
303
310 void stepProcess(int startPoint, int sampleCount)
311 {}
312
318 void triggerProcess(int triggerPoint)
319 {}
320
325 void finishProcess()
326 {}
327
334 void process(int startPoint, int sampleCount)
335 {
336 if (isEnabled())
337 {
338 int currentPoint = startPoint;
339 int stepSize = THIS.startProcess(startPoint, sampleCount);
340 bool triggerActive = samplesToNextTrigger > -1;
341 int samplesToProcessNow = (triggerActive ?
342 std::min(sampleCount, samplesToNextTrigger) :
343 sampleCount);
344
345 do
346 {
347 while (stepSize < samplesToProcessNow)
348 {
349 THIS.stepProcess(currentPoint, stepSize);
350 currentPoint += stepSize;
351 sampleCount -= stepSize;
352 samplesToProcessNow -= stepSize;
353 if (triggerActive) samplesToNextTrigger -= stepSize;
354 }
355
356 if (triggerActive && samplesToNextTrigger < stepSize)
357 {
358 stepProcess(currentPoint, samplesToNextTrigger);
359 currentPoint += samplesToNextTrigger;
360 sampleCount -= samplesToNextTrigger;
361 samplesToNextTrigger = -1;
362 THIS.triggerProcess(currentPoint);
363 triggerActive = samplesToNextTrigger > -1;
364 samplesToProcessNow = (triggerActive ?
365 std::min(sampleCount, samplesToNextTrigger) :
366 sampleCount);
367 }
368 }
369 while (samplesToProcessNow < sampleCount);
370
371 THIS.stepProcess(currentPoint, sampleCount);
372 if (triggerActive) samplesToNextTrigger -= sampleCount;
373 THIS.finishProcess();
374 }
375 }
376};
