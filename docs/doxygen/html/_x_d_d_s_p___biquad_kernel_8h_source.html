87 void setCoefficients()
88 {
89 double w = fastBoundary(cF, 10., 22000.);
90 w *= dspParam.sampleInterval();
91
92 double Q = fastBoundary(qF, 0.1, 10.);
93 Q = cascade ? sqrt(Q) : Q;
94
95 double G = cascade ? sqrt(g) : g;
96
97 double norm;
98
99 if (m == AllPass)
100 {
101 double sinw {sin(2 * M_PI * w)};
102 double cosw {cos(2 * M_PI * w)};
103 double alpha {sinw / (2.0 * Q)};
104 norm = 1.0 / (1.0 + alpha);
105 a2 = b0 = (1.0 - alpha) * norm;
106 b1 = a1 = -2.0 * cosw * norm;
107 b2 = 1.0;
108 }
109 else
110 {
111 double K = tan(M_PI * w);
112
113 switch (m)
114 {
115 default:
116 m = LowPass;
117
118 case LowPass:
119 norm = 1.0 / (1.0 + K / Q + K * K);
120 b0 = b2 = K * K * norm;
121 b1 = 2 * b0;
122 a1 = 2 * (K * K - 1.0) * norm;
123 a2 = (1.0 - K / Q + K * K) * norm;
124 break;
125
126 case HighPass:
127 norm = 1.0 / (1.0 + K / Q + K * K);
128 b0 = b2 = norm;
129 b1 = -2.0 * b0;
130 a1 = 2.0 * (K * K - 1.0) * norm;
131 a2 = (1.0 - K / Q + K * K) * norm;
132 break;
133
134 case BandPass:
135 norm = 1.0 / (1.0 + K / Q + K * K);
136 b0 = K / Q * norm;
137 b1 = 0;
138 b2 = -b0;
139 a1 = 2.0 * (K * K - 1.0) * norm;
140 a2 = (1.0 - K / Q + K * K) * norm;
141 break;
142
143 case Notch:
144 norm = 1.0 / (1.0 + K / Q + K * K);
145 b2 = b0 = (1.0 + K * K) * norm;
146 b1 = 2.0 * (K * K - 1.0) * norm;
147 a1 = b1;
148 a2 = (1.0 - K / Q + K * K) * norm;
149 break;
150
151 case Parametric:
152 if (invert)
153 {
154 norm = 1.0 / (1.0 + G/Q * K + K * K);
155 b0 = (1.0 + 1.0/Q * K + K * K) * norm;
156 b1 = 2.0 * (K * K - 1.0) * norm;
157 b2 = (1.0 - 1.0/Q * K + K * K) * norm;
158 a1 = b1;
159 a2 = (1.0 - G/Q * K + K * K) * norm;
160 }
161 else
162 {
163 norm = 1.0 / (1.0 + 1.0/Q * K + K * K);
164 b0 = (1.0 + G/Q * K + K * K) * norm;
165 b1 = 2.0 * (K * K - 1.0) * norm;
166 b2 = (1 - G/Q * K + K * K) * norm;
167 a1 = b1;
168 a2 = (1.0 - 1.0/Q * K + K * K) * norm;
169 }
170 break;
171
172 case LowShelf:
173 if (invert)
174 {
175 norm = 1.0 / (1.0 + sqrt(2.0*G) * K + G * K * K);
176 b0 = (1.0 + 1.414213562373095 * K + K * K) * norm;
177 b1 = 2.0 * (K * K - 1.0) * norm;
178 b2 = (1.0 - 1.414213562373095 * K + K * K) * norm;
179 a1 = 2.0 * (G * K * K - 1.0) * norm;
180 a2 = (1.0 - sqrt(2.0*G) * K + G * K * K) * norm;
181 }
182 else
183 {
184 norm = 1.0 / (1.0 + 1.414213562373095 * K + K * K);
185 b0 = (1.0 + sqrt(2.0*G) * K + G * K * K) * norm;
186 b1 = 2.0 * (G * K * K - 1.0) * norm;
187 b2 = (1.0 - sqrt(2.0*G) * K + G * K * K) * norm;
188 a1 = 2.0 * (K * K - 1.0) * norm;
189 }
190 break;
191
192 case HighShelf:
193 if (invert)
194 {
195 norm = 1.0 / (G + sqrt(2.0 * G) * K + K * K);
196 b0 = (1.0 + 1.414213562373095 * K + K * K) * norm;
197 b1 = 2.0 * (K * K - 1.0) * norm;
198 b2 = (1.0 - 1.414213562373095 * K + K * K) * norm;
199 a1 = 2.0 * (K * K - G) * norm;
200 a2 = (G - sqrt(2.0 * G) * K + K * K) * norm;
201 }
202 else
203 {
204 norm = 1.0 / (1.0 + 1.414213562373095 * K + K * K);
205 b0 = (G + sqrt(2.0 * G) * K + K * K) * norm;
206 b1 = 2.0 * (K * K - G) * norm;
207 b2 = (G - sqrt(2.0 * G) * K + K * K) * norm;
208 a1 = 2.0 * (K * K - 1.0) * norm;
209 a2 = (1 - 1.414213562373095 * K + K * K) * norm;
210 }
211 break;
212
213 case Custom:
214 break;
215 }
216 }
217 }

490 std::complex<double> filterResponseAtHz(SampleType hz)
491 {
492 const double w = 2.*M_PI*hz*dspParam.sampleInterval();
493 const std::complex<double> mi{0, -1};
494 const std::complex<double> exp1 = std::exp(mi*w);
495 const std::complex<double> exp2 = std::exp(2.*mi*w);
496 const std::complex<double> numerator = b0 + b1*exp1 + b2*exp2;
497 const std::complex<double> denominator = 1. + a1*exp1 + a2*exp2;
498 const std::complex<double> result = numerator/denominator;
499 if (cascade) return result*result;
500 return result;
501 }
502
509 SampleType calculateMagnitudeResponseAtHz(SampleType hz)
510 {
511 return std::abs(filterResponseAtHz(hz));
512 }

529class BiquadFilterKernel
530{
531 SampleType d1 {0.};
532 SampleType d2 {0.};
533 SampleType d3 {0.};
534 SampleType d4 {0.};
535public:
536
541 void reset()
542 {
543 d1 = d2 = d3 = d4 = 0.;
544 }
545
553 SampleType process(const BiquadFilterCoefficients &coeff, SampleType xn)
554 {
555 SampleType s, t;
556 if (coeff.cascade)
557 {
558 /*
559 s = coeff.b0 * xn + d1;
560 d1 = coeff.b1 * xn - coeff.a1 * s + d2;
561 d2 = coeff.b2 * xn - coeff.a2 * s;
562
563 t = coeff.b0 * s + d3;
564 d3 = coeff.b1 * s - coeff.a1 * t + d4;
565 d4 = coeff.b2 * s - coeff.a2 * t;
566 */
567 s = std::fma(coeff.b0, xn, d1);
568 d1 = std::fma(coeff.b1, xn, std::fma(-coeff.a1, s, d2));
569 d2 = std::fma(coeff.b2, xn, -coeff.a2*s);
570
571 t = std::fma(coeff.b0, s, d3);
572 d3 = std::fma(coeff.b1, s, std::fma(-coeff.a1, t, d4));
573 d4 = std::fma(coeff.b2, s, -coeff.a2*t);
574 }
575 else
576 {
577 /*
578 t = coeff.b0 * xn + d1;
579 d1 = coeff.b1 * xn - coeff.a1 * t + d2;
580 d2 = coeff.b2 * xn - coeff.a2 * t;
581 */
582 t = std::fma(coeff.b0, xn, d1);
583 d1 = std::fma(coeff.b1, xn, std::fma(-coeff.a1, t, d2));
584 d2 = std::fma(coeff.b2, xn, -coeff.a2*t);
585 }
586 return t;
587 }
588};
