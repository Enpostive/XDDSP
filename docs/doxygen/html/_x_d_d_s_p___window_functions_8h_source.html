212template <typename WindowType, typename T = SampleType>
213void applyWindowFunction(WindowType window, T* data, unsigned long length)
214{
215 for (unsigned long i = 0; i < length; ++i) data[i] *= window(i);
216}
217
227template <typename WindowType, typename T, unsigned long length>
228void applyWindowFunction(WindowType window, std::array<T, length> &data)
229{ applyWindowFunction(window, data.data(), data.size()); }
