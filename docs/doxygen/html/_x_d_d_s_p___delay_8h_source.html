40template <
41typename SignalIn,
42typename DelayTimeIn,
43typename BufferType = DynamicCircularBuffer<>
44>
45class LowQualityDelay :
46public Component<LowQualityDelay<SignalIn, DelayTimeIn, BufferType>>
47{
48 static_assert(DelayTimeIn::Count == 1, "DelayTimeIn is expected to have just one channel");
49
50 // Private data members here
51 std::array<BufferType, SignalIn::Count> buffer;
52
53public:
54 static constexpr int Count = SignalIn::Count;
55
56 // The input signal
57 SignalIn signalIn;
58
59 // The delay time signal
60 DelayTimeIn delayTimeIn;
61
62 // The delayed signal output.
63 Output<Count> signalOut;
64
65 LowQualityDelay(Parameters &p, SignalIn signalIn, DelayTimeIn delayTimeIn) :
66 signalIn(signalIn),
67 delayTimeIn(delayTimeIn),
68 signalOut(p)
69 {}
70
71 void reset()
72 {
73 for (auto& b : buffer) b.reset(0.);
74 signalOut.reset();
75 }
76
84 void setMaximumDelayTime(uint32_t maxDelay)
85 {
86 for (auto& b : buffer) b.setMaximumLength(maxDelay);
87 }
88
89 void stepProcess(int startPoint, int sampleCount)
90 {
91 for (int i = startPoint, s = sampleCount; s--; ++i)
92 {
93 uint32_t delayTime = fastBoundary(delayTimeIn(0, i), 1., buffer[0].getSize());
94 for (int c = 0; c < Count; ++c)
95 {
96 buffer[c].tapIn(signalIn(c, i));
97 signalOut.buffer(c, i) = buffer[c].tapOut(delayTime);
98 }
99 }
100 }
101};

216 SignalIn signalIn;

222 Output<Count> signalOut;

224 MediumQualityDelay(Parameters &p, SignalIn signalIn, DelayTimeIn delayTimeIn) :
225 signalIn(signalIn),
226 delayTimeIn(delayTimeIn),
227 signalOut(p)
228 {}
229
230 void reset()
231 {

243 void setMaximumDelayTime(uint32_t maxDelay)
244 {

246 }

248 void stepProcess(int startPoint, int sampleCount)

252 const SampleType fDelay = fastBoundary(delayTimeIn(0, i), 1., buffer[0].getSize());
253 IntegerAndFraction iaf(fDelay);
254

257 buffer[c].tapIn(signalIn(c, i));
258 SampleType x0 = buffer[c].tapOut(iaf.intRep());

260 signalOut.buffer(c, i) = LERP(iaf.fracPart(), x0, x1);
261 }
262 }
263 }
