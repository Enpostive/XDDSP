149template <typename T = SampleType>

253template <typename T = SampleType>
254class ModulusCircularBuffer
255{
256 uint32_t size {1};
257
258 std::vector<T> buffer;
259 uint32_t bc;
260public:
265 ModulusCircularBuffer()
266 {
267 setMaximumLength(32);
268 }
269
275 uint32_t getSize() const { return size; }
276
282 void setMaximumLength(uint32_t s)
283 {
284 size = s;
285
286 buffer.resize(size);
287 buffer.shrink_to_fit();
288
289 bc = 0;
290 }
291
297 void reset(T fill)
298 {
299 buffer.assign(size, fill);
300 bc = 0;
301 }
302
309 T tapIn(T input)
310 {
311 bc = (bc + 1) % size;
312 buffer[bc] = input;
313 return input;
314 }
315
323 T tapOut(uint32_t delay) const
324 {
325 if (delay >= size) delay = size - 1;
326 return buffer[(bc - delay + size) % size];
327 }
328
337 T& tapOut(uint32_t delay)
338 {
339 if (delay >= size) delay = size - 1;
340 return buffer[(bc - delay + size) % size];
341 }
342
349 T oneTapRun(T input)
350 {
351 buffer[bc] = input;
352 bc = (bc + 1) % size;
353 return buffer[bc];
354 }
355};
