113template <int LookupTableSize = 512>
114class WaveshapeLookupTable
115{
116 std::array<SampleType, LookupTableSize> lookup;
117 SampleType min;
118 SampleType max;
119 SampleType pointScale;
120
121public:
122 WaveshapeLookupTable()
123 {
124 setTable(-1., 1., [](SampleType x){ return x; });
125 }
126
127 void setTable(SampleType tableMinimum, SampleType tableMaximum, WaveformFunction f)
128 {
129 min = tableMinimum;
130 max = tableMaximum;
131 pointScale = static_cast<SampleType>(LookupTableSize)/(max - min);
132 SampleType pointSize = (max - min)/static_cast<SampleType>(LookupTableSize);
133
134 for (int i = 0; i < LookupTableSize; ++i)
135 {
136 SampleType x = min + pointSize*i;
137 lookup[i] = f(x);
138 x += pointScale;
139 }
140 }
141
142 SampleType operator()(SampleType x)
143 {
144 SampleType t = (x - min)*pointScale;
145 t = fastBoundary(t, 0., static_cast<SampleType>(LookupTableSize - 1));
146
147 IntegerAndFraction tIF(t);
148
149 SampleType step1 = lookup[tIF.intRep()];
150 if (tIF.fracPart() == 0.) return step1;
151 else
152 {
153 SampleType step2 = lookup[tIF.intRep() + 1];
154 return LERP(tIF.fracPart(), step1, step2);
155 }
156 }
157};
