44class RandomNumberBuffer
45{
46public:
47 static constexpr PowerSize NoiseBufferSize {19};
48 static int r;
49
50private:
51 static bool noiseBufferValid;
52 static std::array<SampleType, NoiseBufferSize.size()> noiseBuffer;
53
54public:
60 RandomNumberBuffer()
61 {
62 if (!noiseBufferValid)
63 {
64 noiseBufferValid = true;
65 std::mt19937 gnr; // Construct random number generator with default seed
66 std::uniform_real_distribution<SampleType> rnd(-1., 1.);
67
68 for (SampleType &s: noiseBuffer) s = rnd(gnr);
69
70 auto timeValue = std::chrono::system_clock::now();
71 std::chrono::duration<double> timeDuration = timeValue.time_since_epoch();
72 std::mt19937::result_type seedValue = static_cast<std::mt19937::result_type>(std::chrono::duration_cast<std::chrono::nanoseconds>(timeDuration).count());
73
74 gnr.seed(seedValue);
75 std::uniform_int_distribution<int> iRnd(0, NoiseBufferSize.size());
76 r = iRnd(gnr);
77 }
78 }
79
86 SampleType lookupRandomNumber(int index)
87 {
88 return noiseBuffer[index & NoiseBufferSize.mask()];
89 }
90
96 SampleType lookupNextRandomNumber()
97 {
98 SampleType result = lookupRandomNumber(r);
99 r = (r + 1) & NoiseBufferSize.mask();
100 return result;
101 }
102};

163template <int ChannelCount = 1, int SpectrumSize = 5>
164class PinkNoiseGenerator : public Component<PinkNoiseGenerator<ChannelCount, SpectrumSize>>
165{
166 const PowerSize Size {SpectrumSize};
167 RandomNumberBuffer noise;
168 std::array<SampleType, ChannelCount> accum {0.};
169 std::array<std::array<SampleType, SpectrumSize>, ChannelCount> gnr;
170 int counter {0};
171 int gIndex {0};
172 const SampleType Atten {static_cast<SampleType>(1./sqrt(SpectrumSize))};
173
174public:
175 Output<ChannelCount> noiseOut;
176
177 PinkNoiseGenerator(Parameters &p) :
178 noiseOut(p)
179 {
180 accum.fill(0.);
181 for (auto &g : gnr) g.fill(0.);
182 }
183
184 void reset()
185 {
186 accum.fill(0.);
187 for (auto &g : gnr) g.fill(0.);
188 counter = 0;
189 gIndex = 0;
190 noiseOut.reset();
191 }
192
193 void stepProcess(int startPoint, int sampleCount)
194 {
195 for (int i = startPoint, s = sampleCount; s--; ++i)
196 {
197 counter = (counter + 1) & Size.mask();
198 gIndex = lowestBitSet(counter);
199 for (int c = 0; c < ChannelCount; ++c)
200 {
201 accum[c] -= gnr[c][gIndex];
202 gnr[c][gIndex] = noise.lookupNextRandomNumber();
203 accum[c] += gnr[c][gIndex];
204 noiseOut.buffer(c, i) = accum[c]*Atten;
205 }
206 }
207 }
208};

282 AnalogNoiseSimulator(Parameters &p, SignalIn _signalIn) :
283 signalIn(_signalIn),
284 flickerNoise(p),
285 shotNoise(p),
286 jnNoise(p),
287 snAmp(p, {signalIn}),
288 shotNoiseModulator(p, {shotNoise.noiseOut}, {snAmp.signalOut}),
289 shotNoiseAtten(p, {shotNoiseModulator.signalOut}, {0.001}),
290 jnNoiseModulator(p, {jnNoise.noiseOut}, {signalIn}),
291 whiteNoiseAtten(p, {{shotNoiseAtten.signalOut}, {jnNoiseModulator.signalOut}}, {dB2Linear(-5.)}),

301 jnNoise.reset();
302 snAmp.reset();

305 jnNoiseModulator.reset();
306 whiteNoiseAtten.reset();
307 noiseLevel.reset();
308 }

312 flickerNoise.process(startPoint, sampleCount);
313 shotNoise.process(startPoint, sampleCount);
314 jnNoise.process(startPoint, sampleCount);
315 snAmp.process(startPoint, sampleCount);
316 shotNoiseModulator.process(startPoint, sampleCount);
317 shotNoiseAtten.process(startPoint, sampleCount);
318 jnNoiseModulator.process(startPoint, sampleCount);
319 whiteNoiseAtten.process(startPoint, sampleCount);

321 }
