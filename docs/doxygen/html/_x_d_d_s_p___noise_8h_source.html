39class RandomNumberBuffer
40{
41public:
42 static constexpr PowerSize NoiseBufferSize {19};
43 static int r;
44
45private:
46 static bool noiseBufferValid;
47 static std::array<SampleType, NoiseBufferSize.size()> noiseBuffer;
48
49public:
50 RandomNumberBuffer()
51 {
52 if (!noiseBufferValid)
53 {
54 noiseBufferValid = true;
55 std::mt19937 gnr; // Construct random number generator with default seed
56 std::uniform_real_distribution<SampleType> rnd(-1., 1.);
57
58 for (SampleType &s: noiseBuffer) s = rnd(gnr);
59
60 auto timeValue = std::chrono::system_clock::now();
61 std::chrono::duration<double> timeDuration = timeValue.time_since_epoch();
62 std::mt19937::result_type seedValue = static_cast<std::mt19937::result_type>(std::chrono::duration_cast<std::chrono::nanoseconds>(timeDuration).count());
63
64 gnr.seed(seedValue);
65 std::uniform_int_distribution<int> iRnd(0, NoiseBufferSize.size());
66 r = iRnd(gnr);
67 }
68 }
69
70 SampleType lookupRandomNumber(int index)
71 {
72 return noiseBuffer[index & NoiseBufferSize.mask()];
73 }
74
75 SampleType lookupNextRandomNumber()
76 {
77 SampleType result = lookupRandomNumber(r);
78 r = (r + 1) & NoiseBufferSize.mask();
79 return result;
80 }
81};

138// Increasing the SpectrumSize argument adds lower frequency components to the noise
139template <int ChannelCount = 1, int SpectrumSize = 5>
140class PinkNoiseGenerator : public Component<PinkNoiseGenerator<ChannelCount, SpectrumSize>>
141{
142 // Private data members here
143 const PowerSize Size {SpectrumSize};
144 RandomNumberBuffer noise;
145 std::array<SampleType, ChannelCount> accum {0.};
146 std::array<std::array<SampleType, SpectrumSize>, ChannelCount> gnr;
147 int counter {0};
148 int gIndex {0};
149// const SampleType Atten {1./SpectrumSize};
150 const SampleType Atten {static_cast<SampleType>(1./sqrt(SpectrumSize))};
151
152public:
153 // Specify your inputs as public members here
154 // No inputs
155
156 // Specify your outputs like this
157 Output<ChannelCount> noiseOut;
158
159 // Include a definition for each input in the constructor
160 PinkNoiseGenerator(Parameters &p) :
161 noiseOut(p)
162 {
163 accum.fill(0.);
164 for (auto &g : gnr) g.fill(0.);
165 }
166
167 // This function is responsible for clearing the output buffers to a default state when
168 // the component is disabled.
169 void reset()
170 {
171 accum.fill(0.);
172 for (auto &g : gnr) g.fill(0.);
173 counter = 0;
174 gIndex = 0;
175 noiseOut.reset();
176 }
177
178 void stepProcess(int startPoint, int sampleCount)
179 {
180 for (int i = startPoint, s = sampleCount; s--; ++i)
181 {
182 counter = (counter + 1) & Size.mask();
183 gIndex = lowestBitSet(counter);
184 for (int c = 0; c < ChannelCount; ++c)
185 {
186 accum[c] -= gnr[c][gIndex];
187 gnr[c][gIndex] = noise.lookupNextRandomNumber();
188 accum[c] += gnr[c][gIndex];
189 noiseOut.buffer(c, i) = accum[c]*Atten;
190 }
191 }
192 }
193};
