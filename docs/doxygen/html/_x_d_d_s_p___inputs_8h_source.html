412struct AudioPropertiesInputMode
413{
414 enum InputMode
415 {
416 BeatsPerMinute, // Reports the song tempo in BPM
417 LengthQuarterNoteSeconds, // Reports the length of a quarter note in seconds
418 LengthQuarterNoteSamples, // Reports the length of a quarter note in samples
419 FrequencyQuaterNote, // Reports the song tempo in Hz

432template <int TimeMode, int ChannelCount = 1>
433class AudioPropertiesInput final : public Coupler<AudioPropertiesInput<TimeMode, ChannelCount>, ChannelCount>
434{
435 static constexpr SampleType PerMinute = 1./60.;
436
437 Parameters &dspParam;
438 SampleType multiplier {1.};
439
440public:
448 SampleType get(int channel, int index)
449 {
450 switch (TimeMode)
451 {
452 default:
453 case AudioPropertiesInputMode::BeatsPerMinute:
454 return dspParam.getTempo()*multiplier;
455
456 case AudioPropertiesInputMode::LengthQuarterNoteSeconds:
457 return 60.*multiplier/dspParam.getTempo();
458
459 case AudioPropertiesInputMode::LengthQuarterNoteSamples:
460 return 60.*dspParam.sampleRate()*multiplier/dspParam.getTempo();
461
462 case AudioPropertiesInputMode::FrequencyQuaterNote:
463 return multiplier*dspParam.getTempo()*PerMinute;
464
465 case AudioPropertiesInputMode::SamplesPerSecond:
466 return multiplier*dspParam.sampleRate();
467
468 case AudioPropertiesInputMode::SecondsPerSample:
469 return multiplier*dspParam.sampleInterval();
470 }
471 }
472
473 static constexpr int Count = ChannelCount;
474
480 AudioPropertiesInput(Parameters &p) : dspParam(p)
481 {}
482
489 AudioPropertiesInput(Parameters &p, SampleType _m) : dspParam(p), multiplier(_m)
490 {}
491
497 AudioPropertiesInput(const AudioPropertiesInput<TimeMode> &rhs) :
498 dspParam(rhs.dspParam),
499 multiplier(rhs.multiplier)
500 {}
501
507 SampleType getMultiplier()
508 { return multiplier;}
509
515 void setMultiplier(SampleType _m)
516 { multiplier = _m; }
517};

534template <int NoConnections, int ChannelCount = 1>
535class MultiIn
536{
537protected:
538 using Getter = SampleType(*)(void*, int, int);
539
540 struct Conn
541 {
542 void* self {nullptr};
543 Getter getter {nullptr};
544
545 inline SampleType operator()(int ch, int i) const
546 {
547 return getter ? getter(self, ch, i) : 0.0;
548 }
549 };
550
551 std::array<Conn, NoConnections> connections;
552
553public:
554 static constexpr int Count = ChannelCount;
555
563 template <typename... Inputs>
564 explicit MultiIn(Inputs&... ins)
565 {
566 static_assert(sizeof...(Inputs) == NoConnections,
567 "Must provide exactly NoConnections inputs");
568 size_t idx = 0;
569 (connect(idx++, ins), ...);
570 }
571
572 template <typename C>
573 void connect(size_t idx, C& c)
574 {
575 static_assert(C::Count == ChannelCount,
576 "Channel count mismatch in MultiIn::connect");
577 connections[idx].self = &c;
578 connections[idx].getter = [](void* obj, int ch, int i) -> SampleType
579 {
580 return (*static_cast<C*>(obj))(ch, i);
581 };
582 }
583
584 const std::array<Conn, NoConnections>& getConnections() const
585 {
586 return connections;
587 }
588};

803template <typename Source, typename BufferSampleType, int ChannelCount, int Quality = ProcessQuality::LowQuality>
804class SamplePlaybackHead final : public Coupler<SamplePlaybackHead<Source, BufferSampleType, ChannelCount, Quality>, ChannelCount>
805{
806 static_assert(Quality == ProcessQuality::LowQuality ||
807 Quality == ProcessQuality::MidQuality ||
808 Quality == ProcessQuality::HighQuality,
809 "Invalid quality specifier");
810
811 using size_t = std::size_t;
812
813 std::array<const BufferSampleType*, ChannelCount> buffer;
814 size_t bufferSize {0};
815 SampleType bufferLength {static_cast<SampleType>(bufferSize)};
816
817 Source &input;
818
819public:
827 SampleType get(int channel, int index)
828 {
829 dsp_assert(channel >= 0 && channel < ChannelCount);
830 if (buffer[channel] == nullptr) return 0.;
831
832 SampleType position = fastBoundary(input(channel, index), 0., bufferLength);
833
834 IntegerAndFraction<size_t> posIF(position);
835 size_t xm1, x1, x2;
836
837 switch (Quality)
838 {
839 case ProcessQuality::LowQuality:
840 return buffer[channel][posIF.intRep()];
841 break;
842
843 case ProcessQuality::MidQuality:
844 x1 = std::min(posIF.intRep() + 1, bufferSize);
845 return LERP(posIF.fracPart(),
846 buffer[channel][posIF.intRep()],
847 buffer[channel][x1]);
848
849 case ProcessQuality::HighQuality:
850 xm1 = posIF.intRep() - (posIF.intRep() != 0);
851 x1 = std::min(posIF.intRep() + 1, bufferSize);
852 x2 = std::min(posIF.intRep() + 2, bufferSize);
853 return hermite(posIF.fracPart(),
854 buffer[channel][xm1],
855 buffer[channel][posIF.intRep()],
856 buffer[channel][x1],
857 buffer[channel][x2]);
858 }
859
860 return 0.;
861 }
862
863public:
864 static constexpr int Count = ChannelCount;
865
866 SamplePlaybackHead(Source &_input) :
867 input(_input)
868 {
869 buffer.fill(nullptr);
870 }
871
878 void connectChannel(int channel, const BufferSampleType *ptr)
879 {
880 dsp_assert(channel >= 0 && channel < ChannelCount);
881 buffer[channel] = ptr;
882 }
883
889 std::size_t length()
890 { return bufferSize + 1; }
891
897 void setLength(std::size_t lengthSamples)
898 {
899 bufferSize = lengthSamples - 1;
900 bufferLength = static_cast<SampleType>(bufferSize);
901 }
902};
