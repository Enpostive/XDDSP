46template <typename SignalIn, unsigned long ReserveSize = 32768>
47class LUFSBlockCollector : public Component<LUFSBlockCollector<SignalIn>>, public Parameters::ParameterListener
48{
49 static inline SampleType LUFSDB(SampleType x)
50 { return -0.691 + 10*log10(x); }
51
52 mutable std::mutex mux;
53
54 std::vector<SampleType> blocks;
55 mutable std::vector<SampleType> blockRecord;
56 DynamicCircularBuffer<> buffer;
57 SampleType accum {0.};
58 SampleType recipBlockLength;
59 unsigned int count {0};
60 unsigned int blockInterval {0};
61 unsigned int blockLength {0};
62
63public:
64 static constexpr int Count = SignalIn::Count;
65
70 SignalIn signalIn;
71
78 LUFSBlockCollector(Parameters &p, SignalIn _signalIn) :
79 Parameters::ParameterListener(p),
80 signalIn(_signalIn)
81 {
82 blocks.reserve(ReserveSize);
83 blockRecord.reserve(ReserveSize);
84 updateSampleRate(p.sampleRate(), p.sampleInterval());
85 }
86
87 virtual void updateSampleRate(double sr, double isr) override
88 {
89 blockInterval = 0.1*sr;
90 blockLength = 4*blockInterval;
91 recipBlockLength = 1./blockLength;
92 buffer.setMaximumLength(blockLength);
93 }
94
95 void reset() override
96 {
97 accum = 0.;
98 count = 0;
99 buffer.reset(0.);
100 blocks.clear();
101 }
102
108 int getBlockCount() const
109 { return blocks.size(); }
110
116 SampleType getLastBlock() const
117 {
118 std::lock_guard lock(mux);
119 if (blocks.size() > 0) return LUFSDB(blocks.back());
120 return 0.;
121 }
122
128 SampleType integrateBlocks() const
129 {
130 {
131 std::lock_guard lock(mux);
132 blockRecord = blocks;
133 }
134
135 int acceptedBlocks = 0;
136 SampleType accum = 0;
137
138 for (SampleType i : blockRecord)
139 {
140 if (LUFSDB(i) > -70.)
141 {
142 accum += i;
143 ++acceptedBlocks;
144 }
145 }
146
147 SampleType thresh;
148 if (acceptedBlocks > 0)
149 thresh = LUFSDB(accum/acceptedBlocks) - 10.;
150 else
151 thresh = -70.;
152 acceptedBlocks = 0;
153 accum = 0.;
154
155 for (SampleType i : blockRecord)
156 {
157 if (LUFSDB(i) > thresh)
158 {
159 accum += i;
160 ++acceptedBlocks;
161 }
162 }
163
164 if (acceptedBlocks > 0)
165 thresh = LUFSDB(accum/acceptedBlocks);
166 else
167 thresh = LUFSDB(0.);
168 return thresh;
169 }
170
171
172 void stepProcess(int startPoint, int sampleCount) override
173 {
174 std::lock_guard lock(mux);
175 for (int i = startPoint, s = sampleCount; s--; ++i)
176 {
177 SampleType sqrAccum = 0.;
178 for (int c = 0; c < Count; ++c)
179 {
180 SampleType sqr = signalIn(c, i);
181 sqr *= sqr;
182 sqrAccum += sqr;
183 }
184 accum += buffer.tapIn(sqrAccum);
185 accum -= buffer.tapOut(blockLength);
186
187 ++count;
188 if (count >= blockInterval)
189 {
190 count = 0;
191 blocks.push_back(accum*recipBlockLength);
192 }
193 }
194 }
195};
