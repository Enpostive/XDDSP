175template <typename InternalComponent, int ComponentCount>
176class SummingArray : public Component<SummingArray<InternalComponent, ComponentCount>>
177{
178 std::array<InternalComponent, ComponentCount> components;
179
180public:
181 static constexpr int Count = InternalComponent::Count;
182 static constexpr int CountComponents = ComponentCount;
183
184 Output<Count> sumOut;
185
186 SummingArray(Parameters &p) :
187 components(makeComponentArray<ComponentCount, InternalComponent>(p)),
188 sumOut(p)
189 {}
190
191 void reset()
192 {
193 for (auto &c : components) c.reset();
194 sumOut.reset();
195 }
196
197 void stepProcess(int startPoint, int sampleCount)
198 {
199 for (auto &j : components) j.process(startPoint, sampleCount);
200
201 for (int i = startPoint, s = sampleCount; s--; ++i)
202 {
203 for (int c = 0; c < Count; ++c)
204 {
205 SampleType sum = 0.;
206 for (auto &j : components) sum += j.signalOut(c, i);
207 sumOut.buffer(c, i) = sum;
208 }
209 }
210 }
211
218 InternalComponent& operator[](unsigned int i)
219 {
220 return components[i];
221 }
222
228 auto begin()
229 {
230 return components.begin();
231 }
232
238 auto end()
239 {
240 return components.end();
241 }
242};

261template <int OutputCount, int RampLengthms = 5>
262class MIDIScheduler : public Component<MIDIScheduler<OutputCount, RampLengthms>>, public Parameters::ParameterListener
263{
264 SampleType smoothFactor;
265
266 std::array<SampleType, OutputCount> value;
267 std::array<SampleType, OutputCount> target;
268
269 struct MIDIEvent
270 {
271 SampleType newValue;
272 int samplePosition;
273 int channel;
274 };
275
276 std::vector<MIDIEvent> schedule;
277
278public:
279 static constexpr int Count = OutputCount;
280
281 Output<Count> signalOut;
282
283 MIDIScheduler(Parameters &p) :
284 Component<MIDIScheduler>(p),
285 Parameters::ParameterListener(p)
286 {
287 schedule.reserve(100);
288 updateSampleRate(p.sampleRate(), p.sampleInterval());
289 }
290
298 void addEvent(int channel, SampleType newValue, int samplePosition)
299 {
300 auto it = schedule.begin();
301 while ((it != schedule.end()) && (it->samplePosition < samplePosition)) ++it;
302 schedule.insert(it, {newValue, samplePosition, channel});
303 }
304
305 void stepProcess(int startPoint, int sampleCount)
306 {
307 auto it = schedule.begin();
308 int i, s;
309
310 for (i = startPoint, s = sampleCount;
311 (it != schedule.end()) && (it->samplePosition < sampleCount) && (s--);
312 ++i)
313 {
314 if (it->samplePosition == i)
315 {
316 target[it->channel] = it->newValue;
317 ++it;
318 }
319 for (int c = 0; c < Count; ++c)
320 {
321 expTrack(value[c], target[c], smoothFactor);
322 signalOut.buffer(c, i) = value[c];
323 }
324 }
325
326 for (; s >= 0; --s, ++i)
327 {
328 for (int c = 0; c < Count; ++c)
329 {
330 expTrack(value[c], target[c], smoothFactor);
331 signalOut(c, i) = value[c];
332 }
333 }
334
335 it = schedule.erase(schedule.begin(), it);
336 }
337
343 void advanceMidiEvents(int sampleCount)
344 {
345 auto it = schedule.begin();
346 while (it != schedule.end())
347 {
348 it->samplePosition -= sampleCount;
349 ++it;
350 }
351 }
352
353 virtual void updateSampleRate(SampleType sr, SampleType isr) override
354 { smoothFactor = expCoef(0.001*RampLengthms*sr); }
355};

424 {

478 void allocateVoiceAndStart(const NoteSchedule &ns)
479 {
480 // Keep track of which voice we are allocating, and how many voices have notes on
481 int allocated = voiceCount;
482 bool isNotesOn {voiceOrder.size() > 0};
483
484 // Check if there is another voice already playing the same note
485 for (int i = 0; i < voiceCount; ++i)
486 {
487 if (voices[i]->noteNumber == ns.note) allocated = i;
488 }
489
490 // If a voice is already playing the note, steal it!
491 if (allocated < voiceCount)
492 {
493 int i;
494 for (i = 0;
495 i < voiceOrder.size() && voiceOrder[i] != allocated;
496 ++i)
497 {}
498
499 if (i < voiceOrder.size()) voiceOrder.erase(voiceOrder.begin() + i);
500 }
501 else
502 {
503 allocated = 0;
504
505 // If there are available voices which are not active, simply get the first available
506 if (voiceOrder.size() < voiceCount)
507 {
508 while (allocated < voiceCount && voices[allocated]->isActive()) ++allocated;
509 if (allocated == voiceCount) allocated = 0;
510 }
511 else
512 {
513 // Otherwise, get the oldest note and steal it
514 // First, find the oldest note that has been released
515 while (allocated < voiceOrder.size() && voices[voiceOrder[allocated]]->noteOn)
516 {
517 ++allocated;
518 }
519 if (allocated == voiceOrder.size()) allocated = 0;
520 int t = allocated;
521 allocated = voiceOrder[allocated];
522 voiceOrder.erase(voiceOrder.begin() + t);
523 }
524 }
525
526 SampleType noteFreq = ns.note;
527 // Do we bend the note? If we are not in glissando mode, we only bend if there are already notes being played
528 bool bendNote = polyParam.glissando() || isNotesOn;
529 voiceOrder.push_back(allocated);
530 SampleType fVel = static_cast<SampleType>(ns.velocity)/127.;
531 if (bendNote)
532 {
533 SampleType ln = lastNote;
534 voices[allocated]->startNote(ns.note, noteFreq, fVel, ln, polyParam.portTimeSamples(), true);
535 }
536 else
537 {
538 voices[allocated]->startNote(ns.note, noteFreq, fVel, noteFreq, 0, true);
539 }
540 // Keep the frequency that we calculated for the next bend
541 if (onNoteOn) onNoteOn(allocated);
542 lastNote = noteFreq;
543 }

570 void startLegatoNote(const NoteSchedule &ns)
571 {
572 SampleType fVel = static_cast<SampleType>(ns.velocity)/127.;
573 if (voiceOrder.size() > 0)
574 {
575 // Note already playing, push new note and bend voice
576 voiceOrder.push_back(ns.note);
577 voices[0]->startNote(ns.note, ns.note, fVel, -1., polyParam.portTimeSamples(), false);
578 }
579 else
580 {
581 // First note, push new note and trigger voice
582 voiceOrder.push_back(ns.note);
583 voices[0]->startNote(ns.note, ns.note, fVel, ns.note, polyParam.portTimeSamples(), true);
584 if (onNoteOn) onNoteOn(0);
585 }
586 }

592 void stopLegatoNote(const NoteSchedule &ns)
593 {
594 // Find note in voiceOrder
595 auto it = std::find(voiceOrder.begin(), voiceOrder.end(), ns.note);
596 if (it == voiceOrder.end())
597 {
598 // Note not found or empty vector....ignore
599 }
600 else if (it == --voiceOrder.end())
601 {
602 // Check to see if it is the only note playing
603 if (voiceOrder.size() == 1)
604 {
605 // Note is only note playing. Stop voice
606 voices[0]->stopNote();
607 voiceOrder.pop_back();
608 if (onNoteOff) onNoteOff(0);
609 }
610 else
611 {
612 // Note is last note played. Pop note and bend voice back to new last note
613 SampleType ln = voices[0]->noteNumber;
614 voiceOrder.pop_back();
615 voices[0]->startNote(voiceOrder.back(),
616 voiceOrder.back(),
617 voices[0]->voiceComponents[0]->velocityIn.getControl(), ln,
618 polyParam.portTimeSamples(),
619 false);
620 }
621 }
622 else
623 {
624 // Note is in the order but it is not currently playing. Remove it from the vector
625 voiceOrder.erase(it);
626 }
627 }

633 void doNoteAction(const NoteSchedule &ns)
634 {
635 switch (ns.velocity)
636 {
637 case NoteSchedule::AllNotesOff:
638 {
639 if (polyParam.legato())
640 {
641 voiceOrder.clear();
642 voices[0]->stopNote();
643 }
644 else
645 {
646 for (auto it: voiceOrder)
647 {
648 voices[it]->stopNote();
649 }
650 }
651 }
652 break;
653
654 case NoteSchedule::AllSoundOff:
655 {
656 resetAllNotes();
657 }
658 break;
659
660 default:
661 {
662 if (polyParam.legato())
663 {
664 if (ns.velocity == 0)
665 {
666 stopLegatoNote(ns);
667 }
668 else
669 {
670 startLegatoNote(ns);
671 }
672 }
673 else
674 {
675 if (ns.velocity == 0)
676 {
677 stopVoice(ns);
678 }
679 else
680 {
681 allocateVoiceAndStart(ns);
682 }
683 }
684 }
685 break;
686 }

859 void process(int startPosition, int sampleCount)
860 {
861 int i = startPosition;
862 int s = sampleCount;
863 while (!schedule.empty() && s > 0)
864 {
865 int ns = schedule[0].samplePosition - i;
866 if (ns > s) ns = s;
867 if (ns > 0)
868 {
869 parallelProcess(i, ns);
870 voiceArray.process(i, ns);
871 s -= ns;
872 i += ns;
873 }
874 if (i == schedule[0].samplePosition)
875 {
876 doNoteAction(schedule[0]);
877 schedule.erase(schedule.begin());
878 }
879 }
880
881 if (s > 0)
882 {
883 parallelProcess(i, s);
884 voiceArray.process(i, s);
885 }
886
887 purgeInactiveVoices();
888 }
