190inline SampleType hermite(SampleType fracPos, SampleType xm1, SampleType x0, SampleType x1, SampleType x2)
191{
192 const SampleType c = 0.5*(x1 - xm1);
193 const SampleType v = x0 - x1;
194 const SampleType w = c + v;
195 const SampleType a = w + v + 0.5*(x2 - x0);
196 const SampleType b_neg = w + a;
197
198 return ((((a*fracPos) - b_neg)*fracPos + c)*fracPos + x0);
199}

353class MinMax
354{
355 SampleType _min {0.};
356 SampleType _max {1.};
357 SampleType _delta {_max - _min};
358
359public:

383 MinMax(const MinMax& rhs) :

385 _max(rhs._max),
386 _delta(rhs._delta)
387 {}
388
395 void setMinMax(SampleType min, SampleType max)
396 {
397 if ((TopBottom == 0) && (min > max)) std::swap(min, max);
398 _min = min;
399 _max = max;
400 _delta = _max - _min;
401 }
402
408 void setMin(SampleType min)
409 { setMinMax(min, _max); }
410
416 void setMax(SampleType max)
417 { setMinMax(_min, max); }
418
424 SampleType min()

426
432 SampleType max()
433 { return _max; }
434
442 SampleType delta()
443 { return _delta; }
444
453 SampleType fastBoundary(SampleType input)
454 {
455 if (TopBottom == 0) return ::XDDSP::fastBoundary(input, _min, _max);
456 return (_min < _max) ?
457 ::XDDSP::fastBoundary(input, _min, _max) :
458 ::XDDSP::fastBoundary(input, _max, _min);
459 }
460
467 SampleType LERP(SampleType input)
468 { return ::XDDSP::LERP(input, _min, _max); }
469

631 void setToNextPowerTwo(uint32_t rs)
632 {
633 static const int MultiplyDeBruijnBitPosition[32] =
634 {
635 0, 9, 1, 10, 13, 21, 2, 29, 11, 14, 16, 18, 22, 25, 3, 30,
636 8, 12, 20, 28, 15, 17, 24, 7, 19, 27, 23, 6, 26, 5, 4, 31
637 };
638
639 m = nextPowerTwoMinusOne(rs);
640 b = MultiplyDeBruijnBitPosition[(uint32_t)(m * 0x07C4ACDDU) >> 27];
641 s = m + 1;
642 }

650 static uint32_t nextPowerTwoMinusOne(uint32_t rs)
651 {
652 rs = rs - 1;
653 rs |= rs >> 1;
654 rs |= rs >> 2;
655 rs |= rs >> 4;
656 rs |= rs >> 8;
657 rs |= rs >> 16;
658 return rs;
659 }

783class IntersectionEstimator
784{
785 static constexpr double c1_6 = 1./6.;
786 static constexpr double c3_2 = 3./2.;
787 static constexpr double c5_2 = 5./2.;
788 static constexpr double c1_3 = 1./3.;
789 static constexpr double c11_6 = 11./6.;
790
791 double a, b, c, d, e, mu;
792
793 double f(double x)
794 {
795 double xx = x * x;
796 return a*xx*x + b*xx + c*x + e;
797 }
798
799public:
810 void setSampleValues(SampleType xm2,
811 SampleType xm1,
812 SampleType x1,
813 SampleType x2)
814 {
815 a = -c1_6*xm2 + 0.5*xm1 - 0.5*x1 + c1_6*x2;
816 b = xm2 - c5_2*xm1 + 2*x1 - 0.5*x2;
817 c = -c11_6*xm2 + 3*xm1 - c3_2*x1 + c1_3*x2;
818 e = xm2;
819 }
820
827 void estimateIntersection(double p, double epsilon = 0.001)
828 {
829 double _a = 1.;
830 double _b = 2.;
831 double fa = f(_a) - p;
832 double fb = f(_b) - p;
833 double _c = 1.5;
834 double fc;
835
836 if ((fa < 0. && fb > 0.) || (fa > 0. && fb < 0.))
837 {
838 for (int i = 0; i < 8; ++i)
839 {
840 _c = 0.5*(_a+_b);
841 fc = f(_c) - p;
842 if (std::signbit(fa) == std::signbit(fc))
843 {
844 _a = _c;
845 fa = fc;
846 }
847 else
848 {
849 _b = _c;
850 }
851 }
852 }
853
854
855 double dd = _c;
856
857 double err = (_b - _a) / 2.;
858 double ddd;
859 int looplimit = 4;
860
861 while (looplimit && fabs(err) > epsilon && f(dd) != 0.)
862 {
863 --looplimit;
864 mu = 3.*a*dd*dd + 2.*b*dd + c;
865 if (mu != 0) err = (f(dd) - p)/mu;
866 else err = 0.;
867 ddd = dd + err;
868 // ddd = dd - (a*dd*dd*dd + b*dd*dd + c*dd + e - p)/mu;
869 dd = ddd;
870 }
871
872 if (dd <= 1. || dd >= 2.)
873 {
874 dd = 1.5;
875 }
876
877 d = dd - 1.;
878 }
879
880 // TODO: Document calculateStationaryPoints or deprecate it.
881 bool calculateStationaryPoints(SampleType &minimum, SampleType &maximum)
882 {
883 const double dis = b*b - 3.*a*c;
884 if (dis < 0.) return false;
885
886 const double quadVertex = calculateInflectionPoint();
887 if (dis == 0.)
888 {
889 minimum = quadVertex;
890 maximum = quadVertex;
891 return true;
892 }
893
894 const double sqrtDis = std::sqrt(dis)/(3.*a);
895 minimum = quadVertex - sqrtDis;
896 maximum = quadVertex + sqrtDis;
897
898 if (f(minimum) > f(maximum)) std::swap(minimum, maximum);
899 return true;
900 }
901
902 // TODO: Document calculateInflectionPoint or deprecate it.
903 SampleType calculateInflectionPoint()
904 {
905 return -b/(3.*a);
906 }
907
914 void getIntersectionValues(SampleType &frac, SampleType &slope)
915 {
916 frac = d;
917 slope = mu;
918 }
919};

940template <int Size, int Quality = ProcessQuality::MidQuality>
941class LookupTable
942{
943 static_assert(Quality == ProcessQuality::LowQuality ||
944 Quality == ProcessQuality::MidQuality ||
945 Quality == ProcessQuality::HighQuality,
946 "Invalid quality specifier");
947
948 static constexpr int Padding()
949 {
950 return
951 (Quality == ProcessQuality::LowQuality) ? 1 :
952 ((Quality == ProcessQuality::MidQuality) ? 2 : 4);
953 }
954
955 static constexpr int Offset()
956 {
957 return
958 (Quality == ProcessQuality::LowQuality) ? 0 :
959 ((Quality == ProcessQuality::MidQuality) ? 0 : 1);
960 }
961
962 static constexpr SampleType RecSize = 1./Size;
963
964 std::array<SampleType, Size + Padding()> table;
965
966public:
970 MinMax<> boundaries;
971
976 LookupTable()
977 {
978 table.fill(0.);
979 }
980
990 void calculateTable(WaveformFunction func)
991 {
992 for (int i = 0; i < Size + Padding(); ++i)
993 {
994 SampleType x = boundaries.LERP(static_cast<SampleType>(i - Offset())*RecSize);
995 table[i] = func(x);
996 }
997 }
998
1005 SampleType lookup(SampleType x)
1006 {
1007 SampleType y, ym1, y0, y1, y2;
1008
1009 x = boundaries.fastBoundary(x);
1010 x = boundaries.normalise(x)*Size;
1011 IntegerAndFraction xIF(x);
1012 int i0 = xIF.intRep() + Offset();
1013
1014 switch(Quality)
1015 {
1016 case ProcessQuality::LowQuality:
1017 y = table[i0];
1018 break;
1019
1020 case ProcessQuality::MidQuality:
1021 y0 = table[i0];
1022 y1 = table[i0 + 1];
1023 y = LERP(xIF.fracPart(), y0, y1);
1024 break;
1025
1026 case ProcessQuality::HighQuality:
1027 ym1 = table[i0 - 1];
1028 y0 = table[i0];
1029 y1 = table[i0 + 1];
1030 y2 = table[i0 + 2];
1031 y = hermite(xIF.fracPart(), ym1, y0, y1, y2);
1032 break;
1033 }
1034
1035 return y;
1036 }
1037
1044 SampleType operator()(SampleType x) { return lookup(x); }
1045};
